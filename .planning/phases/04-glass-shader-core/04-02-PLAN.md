    setGlassTint(r: number, g: number, b: number): void;
  } | null;
  destroyEngine(): void;
}

<task type="auto">

  <files>src/App.tsx</files>
  <action>
In `src/App.tsx`, after the engine is initialized and the `ResizeObserver` is set up (after `observerRef.current = observer;` and before `setStatus('running');`), add code to set the initial glass parameters so the glass effect is visible from the first frame:

```typescript
const engine = module.getEngine();
if (engine) {
  // Center glass rectangle: 50% of canvas, centered
  engine.setGlassRect(0.25, 0.25, 0.5, 0.5);
  // Visible defaults: 20px corners, moderate blur, light tint, subtle refraction
  engine.setGlassParams(20.0, 0.5, 0.15, 0.02);
  // White tint
  engine.setGlassTint(1.0, 1.0, 1.0);
}
```

Note: The C++ engine already sets these defaults in `init()` (from Plan 01), but calling them from JS confirms the Embind bridge works and allows future React-side parameter control. This is intentionally redundant -- it validates the full JS->C++ path.
